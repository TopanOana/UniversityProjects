% valleyAspect(l1...ln, OK) = true, if n=1 and OK=0
%                             valleyAspect(l2...ln, 0), if l1<l2

valleyAspect([_],0).

    H1<H2,
    !,
    valleyAspect([H2|T], 0).

    H1>H2,
    valleyAspect([H2|T],1).
